* Setup

#+BEGIN_SRC shell
pip install cs50
touch database.db
#+END_SRC

* Connecting

#+BEGIN_SRC python :tangle main.py :mkdirp yes
from cs50 import SQL

db = SQL("sqlite:///database.db")
#+END_SRC

* CREATE

#+BEGIN_SRC python :tangle main.py
db.execute("""
    CREATE TABLE IF NOT EXISTS users (
        name     TEXT,
        age      NUMBER,
        fav_food STRING
    )
""")
#+END_SRC

* INSERT

Use =?= placeholders — never interpolate values directly into SQL strings.

#+BEGIN_SRC python :tangle main.py
db.execute(
    "INSERT INTO users (name, age, fav_food) VALUES(?, ?, ?)",
    "eesa", 14, "pizza"
)
db.execute(
    "INSERT INTO users (name, age, fav_food) VALUES(?, ?, ?)",
    "bob", 20, "burgers"
)
#+END_SRC

* SELECT

#+BEGIN_SRC python :tangle main.py
# All rows
people = db.execute("SELECT * FROM users")
print(people)

# Distinct values
foods = db.execute("SELECT DISTINCT fav_food FROM users")
print(foods)

# Filter
bob = db.execute("SELECT * FROM users WHERE name='bob'")
print(bob)

# Compound condition
result = db.execute(
    "SELECT * FROM users WHERE name='bob' AND age=20"
)
print(result)
#+END_SRC

* UPDATE

Always include =WHERE= or every row gets overwritten.

#+BEGIN_SRC python :tangle main.py
db.execute(
    "UPDATE users SET fav_food='shawarma' WHERE name='eesa'"
)
#+END_SRC

* DELETE

#+BEGIN_SRC python :tangle main.py
# One row
db.execute("DELETE FROM users WHERE name='bob'")

# All rows
db.execute("DELETE FROM users")
#+END_SRC
